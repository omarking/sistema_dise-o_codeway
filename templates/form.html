Formularios


Entradas de texto o Inputs


Las entradas de texto también llamados inputs, son componentes que ayudan a agregar campos en los formularios.
Estos se pueden utilizar únicamente con texto o acompañarlo con un icono al inicio o al final.
Puede utilizar las clases de bordes contextuales .border-* para cambiar el color del input.

Ejemplo:



<form class="form-floating Q">
    <input type="email" class="form-control border-primary" id="floatingInputValue" placeholder="name@example.com"
        value="test@example.com">
    <label for="floatingInputValue">Input with value</label>
</form>

Estado desactivo


Puede agregar dentro del input la propiedad disabled para desactivar el elemento. 



<form class="form-floating Q">
    <input type="email" class="form-control border-primary" id="floatingInputValue" placeholder="name@example.com"
        value="test@example.com" disabled>
    <label for="floatingInputValue">Input with value</label>
</form>

Input error


Puede agregar dentro del input la propiedad disabled para desactivar el elemento. 



<form class="form-floating Q">
    <input type="email" class="form-control is-invalid" id="floatingInputInvalid" placeholder="name@example.com"
        value="test@example.com">
    <label for="floatingInputInvalid">Invalid input</label>
</form>

Carga de archivos con Previsualizador simple


Este elemento permite subir un documento o imagen y previsualizarla al cargarse, siempre y cuando sea simple y ligero.
Puede modificar la estructura y diseño del documento en el codigo JS. 



<!--Estructura HTML-->
<input id="fileSimple" class="form-control btn-primary" type="file"/>
    <hr>
<div id="previewSimple" style="width:50%;"></div>

<!--Previsualizador JS-->
<script>
    document.getElementById("fileSimple").onchange = function (e) {
        // Creamos el objeto de la clase FileReader
        let reader = new FileReader();

        // Leemos el archivo subido y se lo pasamos a nuestro fileReader
        reader.readAsDataURL(e.target.files[0]);

        // Cuando el archivo este cargado se ejecutara el siguiente código
        reader.onload = function () {
            let preview = document.getElementById('previewSimple'),
                doc = document.createElement('embed');

            //Propiedades elemento cargado (doc)
            doc.width = '100%';
            doc.src = reader.result;

            //Propiedades elemento contenedor del archivo 
            preview.className = 'img-thumbnail';
            preview.innerHTML = '';
            preview.append(doc);
        };
    }
</script>

Carga de archivos complejo


Este elemento permite cargar imagenes con su nombre y extension.
Solo admite una imagen a la vez. 



<!--Estructura HTML-->
<div class="cnt-pre">
    <figure class="image-container">
        <img id="chosen-image">
        <figcaption id="file-name" class="fc"></figcaption>
    </figure>

    <input type="file" id="upload-button" accept="image/*" class="in">
    <label for="upload-button" class="lb btn-primary">
        <ion-icon name="cloud-upload-outline" size="large"></ion-icon>
        <br>
        Elegir imagen
    </label>
</div>


<!--Codigo JS-->
<script>>
    let uploadButton = document.getElementById("upload-button");
    let chosenImage = document.getElementById("chosen-image");
    let fileName = document.getElementById("file-name");

    uploadButton.onchange = () => {
        let reader = new FileReader();
        reader.readAsDataURL(uploadButton.files[0]);
        reader.onload = () => {
            chosenImage.setAttribute("src", reader.result);
        }
        fileName.textContent = uploadButton.files[0].name;
    }
</script>

Carga de archivos complejo y multiple


Este elemento permite cargar imagenes con su nombre y extension.
Permite agregar varias imagenes a la vez. 



<!--Estructura HTML-->
<div class="cnt-mul">
    <input type="file" id="file-input" class="in-mul" accept="image/png, image/jpeg" onchange="preview()" multiple>
    <label for="file-input" class="lb-mul btn-secondary">
        <ion-icon name="cloud-upload-outline" size="large"></ion-icon>
        <br>
        Elegir imagen
    </label>
    <p id="num-of-files" class="p-mul"></p>
    <div id="images-mul"></div>
</div>


<!--Codigo JS-->
<script>
let fileInput = document.getElementById("file-input");
let imageContainer = document.getElementById("images-mul");
let numOfFiles = document.getElementById("num-of-files");

function preview() {
    imageContainer.innerHTML = "";
    numOfFiles.textContent = `${fileInput.files.length} Files Selected`;

    for (i of fileInput.files) {
        let reader = new FileReader();
        let figure = document.createElement("figure");
        figure.className = "figure-mul";
        let figCap = document.createElement("figcaption");
        figCap.className = "figcaption-mul";
        figCap.innerText = i.name;
        figure.appendChild(figCap);
        reader.onload = () => {
            let img = document.createElement("img");
            img.className = "img-mul";
            img.setAttribute("src", reader.result);
            figure.insertBefore(img, figCap);
        }
        imageContainer.appendChild(figure);
        reader.readAsDataURL(i);
    }
}
</script>

<!--Codigo JS-->
<script>
    document.querySelector('#pdffFile').addEventListener('change', () => {

        let pdffFile = document.querySelector('#pdffFile').files[0];

        document.querySelector('#vistaPrevia').setAttribute('src', pdffFileURL);
    })
</script>

DropZone


Este elemento permite cargar archivos, arrastrando y soltando.
Para este ejemplo utilizamos una libreria llamada Dropzone.JS, es necesario referenciar la siguiente CDN, estructura y codigo. 



<!--CDN Libreria-->
<script src="https://unpkg.com/dropzone@5/dist/min/dropzone.min.js"></script>
<link rel="stylesheet" href="https://unpkg.com/dropzone@5/dist/min/dropzone.min.css" type="text/css" />

<!--Estructura HTML-->
<form action="/target" class="dropzone shadow-lg p-3 mb-5 bg-body rounded border-primary"
    id="my-great-dropzone">
</form>

<!--Codigo JS-->
<script>
    Dropzone.options.myGreatDropzone = {
        paramName: "file", // Nombre del archivo
        maxFilesize: 2, // Tamaño maxiomo en MB
        accept: function (file, done) {
            if (file.name == ".jpg") {
                done("Accept");
            } else {
                done();
            }
        }
    };
</script>

<script>
    // Ejemplo de JavaScript de inicio para deshabilitar el envío de formularios si hay campos no válidos
    (function () {
        'use strict'

        // Obtener todos los formularios a los que queremos aplicar estilos de validación personalizados
        var forms = document.querySelectorAll('.needs-validation')

        // Ciclo de busqueda
        Array.prototype.slice.call(forms)
            .forEach(function (form) {
                form.addEventListener('submit', function (event) {
                    if (!form.checkValidity()) {
                        event.preventDefault()
                        event.stopPropagation()
                    }

                    form.classList.add('was-validated')
                }, false)
            })
    })()
</script>
